```bash
mkdir -p /opt/security-tools
cd /opt/security-tools

# Pentesting
git clone --depth 1 https://github.com/rapid7/metasploit-framework
git clone --depth 1 https://github.com/swisskyrepo/PayloadsAllTheThings
git clone --depth 1 https://github.com/danielmiessler/SecLists

# AI Tools
git clone --depth 1 https://github.com/anomalyco/opencode
git clone --depth 1 https://github.com/ollama/ollama
git clone --depth 1 https://github.com/nomic-ai/gpt4all

# Red Team
git clone --depth 1 https://github.com/cobbr/Covenant
git clone --depth 1 https://github.com/BC-SECURITY/Empire

# Scanning
git clone --depth 1 https://github.com/projectdiscovery/nuclei
git clone --depth 1 https://github.com/projectdiscovery/naabu
git clone --depth 1 https://github.com/robertdavidgraham/masscan

# WiFi
git clone --depth 1 https://github.com/derv82/wifite2
git clone --depth 1 https://github.com/FluxionNetwork/fluxion

echo "All repos cloned!"
```
